    this file show you how to publish some data over Choir network

    #include <choir_client.h>
    #include <choir_logger.h>
    #include <stdint.h>
    #include <stdio.h>
    #include <stdlib.h>
    #include <string.h>
    #ifdef WIN32
    #include <windows.h>
    #elif defined(linux)
    #include <unistd.h>
    #include <signal.h>
    #endif
    static volatile int keepRunning = 1;
    void intHandler(int dummy);
    void intHandler(int dummy)
    {
    (void)dummy;
    keepRunning = 0;
    }
    int main(void)
    {
    choir_socket_handler_t* choir_handler;
    signal(SIGINT, intHandler);
    signal(SIGTERM, intHandler);
    setDebug(0);
    int n = 0; // error handler
    int i = 0;
    // Alloc choir_socket_handler_t structure
    choir_handler = choir_alloc();
    if (choir_handler == NULL) {
    printf("choir_alloc error");
    return -1;
    }
    // Connect to the middleware-server with application identifier 1024
    n = choir_connect(choir_handler, "tcp://127.0.0.1:49154", 1024);
    if (n != 0) {
    printf("Choir connect error\n");
    return -1;
    }
    // Initialise the choir_pubsub_message
    struct choir_pubsub_message message = CHOIR_PUBSUB_MESSAGE_INIT;
    message.data = malloc(sizeof(char) * 512);
    if (message.data == NULL) {
    printf("malloc error");
    return -1;
    }
    while (keepRunning) {
    sprintf((char*)message.data, "{ \"brake\" : \"%d\" }", i);
    message.data_length = strlen((char*)message.data);
    // publish message to message id 75
    choir_publish(choir_handler, 75, &message);
    i++;
    sleep(1);
    }
    free(message.data);
    // perform disconnect and release socket
    choir_disconnect(choir_handler);
    // free allocated memory for choir_handler
    choir_free(choir_handler);
    return 0;
    }
