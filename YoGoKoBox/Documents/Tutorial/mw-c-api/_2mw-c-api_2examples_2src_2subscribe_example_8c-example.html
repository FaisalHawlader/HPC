    #include <choir_client.h>
    #include <choir_logger.h>
    #include <stdint.h>
    #include <stdio.h>
    #include <stdlib.h>
    #include <string.h>
    #include <inttypes.h>
    #ifdef WIN32
    #include <windows.h>
    #elif defined(linux)
    #include <unistd.h>
    #include <signal.h>
    #endif
    static volatile int keepRunning = 1;
    void intHandler(int dummy);
    void intHandler(int dummy)
    {
    (void)dummy;
    keepRunning = 0;
    }
    int32_t subscribe_id = 0;
    static void print_distribute_message(struct choir_pubsub_message* message, int32_t subscription_id, void* u_data)
    {
    uint32_t i = 0;
    (void)u_data;
    (void)subscription_id;
    if (subscribe_id == subscription_id) {
    printf("message : \n");
    for (i = 0; i < message->data_length; i++) {
    printf("%2.2x", message->data[i]);
    if (i % 15 == 0 && i != 0) {
    printf("\n");
    }
    }
    printf("\n%s\n", message->data);
    uint64_t j = 0;
    for (i = 0; i < choir_map_length(message->metadata); i++) {
    const char* key = choir_map_key(message->metadata, i);
    uint8_t type = choir_map_type(message->metadata, key);
    size_t size = 0;
    const uint8_t* ptr = NULL;
    switch (type) {
    case MAP_TYPE_NULL:
    printf("%s = NIL", key);
    break;
    case MAP_TYPE_BOOLEAN:
    printf("%s = %s", key, choir_map_get_bool(message->metadata, key, false) ? "true" : "false");
    break;
    case MAP_TYPE_INT:
    printf("%s = %"PRId64, key, choir_map_get_int(message->metadata, key, 0));
    break;
    case MAP_TYPE_DOUBLE:
    printf("%s = %f", key, choir_map_get_double(message->metadata, key, 0.0));
    break;
    case MAP_TYPE_STRING:
    size = 0;
    ptr = (const uint8_t*)choir_map_get_string(message->metadata, key, "empty", &size);
    printf("%s = \"%.*s\"", key, (int)size, ptr);
    break;
    case MAP_TYPE_BIN:
    size = 0;
    ptr = choir_map_get_bin(message->metadata, key, 0, &size);
    printf("%s = [", key);
    for (j = 0; j < size; j++) {
    printf("%02x%c", ptr[j], (j < size - 1) ? ',' : ']');
    }
    break;
    }
    printf("\n");
    }
    } else {
    printf("Receive distribute message with incorrect subscription id\n");
    }
    }
    int main(void)
    {
    choir_socket_handler_t* choir_handler;
    signal(SIGINT, intHandler);
    signal(SIGTERM, intHandler);
    setDebug(0);
    int n = 0;
    // Alloc choir_socket_handler_t structure
    choir_handler = choir_alloc();
    if (choir_handler == NULL) {
    printf("choir_alloc error");
    return -1;
    }
    // Connect to the middleware-server with application identifier 1024
    n = choir_connect(choir_handler, "tcp://127.0.0.1:49154", 1025);
    if (n != 0) {
    printf("Choir connect error\n");
    return -1;
    }
    // Subscribe to corresponding MESSAGE_ID and register callback function
    // "print_distribute_message" that will be called when receiving distribute message
    // return a subscribe_id use when you want to unsubscribe from server
    subscribe_id = choir_subscribe(choir_handler, 75, NULL, print_distribute_message, NULL);
    if (subscribe_id < 0) {
    printf("subscribe error\n");
    goto end_subscribe;
    }
    // check distribute message that correspond to MESSAGE_ID and give them to callback function
    // "print_distribute_message"
    // get only 10 message
    int i = 0;
    for (i = 0; i < 10 && keepRunning; i++) {
    choir_loop_once(choir_handler);
    sleep(1);
    }
    // Unsubscribe from MESSAGE_ID, server should stop to distribute message to this APPLICATION_ID
    n = choir_unsubscribe(choir_handler, subscribe_id);
    end_subscribe:
    // perform disconnect if needed and release socket
    choir_disconnect(choir_handler);
    // free allocated memory for choir_handler
    choir_free(choir_handler);
    return 0;
    }
