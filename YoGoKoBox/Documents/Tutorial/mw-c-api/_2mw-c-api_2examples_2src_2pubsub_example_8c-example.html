    Example of publish and subscribe to the same message ID with choir_loop_once

    #define PUBLISH_APPLICATION_ID 42 // Client-unique ID of this application use anyone you want
    #define SUBSCRIBE_APPLICATION_ID 43 // Client-unique ID of this application use anyone you want
    #define DESTINATION_ID 0x30 // for ipv6 localhost distribution
    #define MESSAGE_ID 100 // Can Bus message id in choir-java , ID that correspond to a type of data source
    #include <choir_client.h>
    #include <choir_logger.h>
    #include <stdint.h>
    #include <stdio.h>
    #include <stdlib.h>
    #include <string.h>
    #include <float.h>
    int32_t subscribe_id = 0;
    static void print_distribute_message(struct choir_pubsub_message* message, int32_t subscription_id, void* u_data)
    {
    uint32_t i = 0;
    (void)u_data;
    (void)subscription_id;
    if (subscribe_id == subscription_id) {
    printf("message hex : \n");
    for (i = 0; i < message->data_length; i++) {
    printf("%2.2x", message->data[i]);
    if (i % 15 == 0 && i != 0) {
    printf("\n");
    }
    }
    printf("\nMessage ascii :\n%s\n", message->data);
    if(choir_map_length(message->metadata)>0){
    printf("\nMetadata : \n");
    choir_map_print(message->metadata);
    }
    printf("\n");
    } else {
    printf("Receive distribute message with incorrect subscription id\n");
    }
    }
    int main(void)
    {
    choir_socket_handler_t* choir_handler_publish;
    choir_socket_handler_t* choir_handler_subscribe;
    int n = 0; // error handler
    struct choir_pubsub_message message = CHOIR_PUBSUB_MESSAGE_INIT;
    int i = 0;
    setDebug(0);
    // Alloc choir_socket_handler_t structure
    choir_handler_publish = choir_alloc();
    choir_handler_subscribe = choir_alloc();
    // Generic connect to server (TCP/UDP)
    n = choir_connect(choir_handler_publish, "tcp://127.0.0.1:49154", PUBLISH_APPLICATION_ID);
    n = choir_connect(choir_handler_subscribe, "tcp://127.0.0.1:49154", SUBSCRIBE_APPLICATION_ID);
    char value[256];
    uint8_t binary[6] = { 0x79, 0x6f, 0x67, 0x6f, 0x6b, 0x6f };
    strcpy(value, "strvalue");
    message.data = malloc(sizeof(char) * 512);
    if (message.data== NULL) {
    printf("malloc error");
    return -1;
    }
    if (n != -1) {
    // Subscribe to corresponding MESSAGE_ID and register callback function
    // "print_distribute_message" that will be called when receiving distribute message
    // return a subscribe_id use when you want to unsubscribe from server
    subscribe_id = choir_subscribe(
    choir_handler_subscribe, MESSAGE_ID, NULL, print_distribute_message, NULL);
    if (subscribe_id < 0) {
    printf("subscribe error\n");
    goto end_subscribe;
    }
    for(i= 0;i<10;i++) {
    sprintf((char*)message.data, "{ \"brake\" : \"%d\" }", i);
    message.data_length = strlen((char*)message.data);
    message.metadata = choir_map_init(10);
    choir_map_put_bool(message.metadata, "trueValue", true);
    choir_map_put_bool(message.metadata, "falseValue", false);
    choir_map_put_int(message.metadata, "int", 2);
    choir_map_put_int(message.metadata, "int_max", INT64_MAX);
    choir_map_put_int(message.metadata, "int_min", INT64_MIN);
    choir_map_put_double(message.metadata, "double", 3);
    choir_map_put_double(message.metadata, "double_max", DBL_MAX);
    choir_map_put_double(message.metadata, "double_min", 0 - DBL_MAX);
    choir_map_put_string(message.metadata, "str_key", value, strlen(value));
    choir_map_put_bin(message.metadata, "binary", binary, 6);
    // publish json string to server by corresponding message ID
    n = choir_publish(choir_handler_publish, MESSAGE_ID, &message);
    if (n != message.data_length) {
    printf("error\n");
    break;
    }
    choir_map_free(message.metadata);
    message.metadata = NULL;
    choir_loop_once(choir_handler_subscribe);
    i++;
    }
    // Unsubscribe from subscrive_id, server should stop to distribute message to this APPLICATION_ID
    n = choir_unsubscribe(choir_handler_subscribe, subscribe_id);
    end_subscribe:
    free(message.data);
    // perform disconnect if needed and release socket
    choir_disconnect(choir_handler_publish);
    choir_disconnect(choir_handler_subscribe);
    // free allocated memory for choir_handler
    choir_free(choir_handler_publish);
    choir_free(choir_handler_subscribe);
    }
    return 0;
    }
