    static void print_help(void)
    {
    printf("mwpubsub :");
    printf("Receive and send messages to choir middleware, with a netcat style\n");
    printf("Usage :\n");
    printf(" mwpubsub -p -a <app_id> -m <msg_id> [option] Publish mode, send messages to "

    printf(" mwpubsub -s -a <app_id> -m <msg_id> [option] Subscribe mode, receive messages from "

    uint8_t hex_flag = 0;
    int32_t subscribe_id;
    uint8_t print_metadata_flag = 0x0;
    static void print_distribute_message(struct choir_pubsub_message* message, int32_t subscription_id,
    void* u_data)
    {
    int i = 0;
    if(subscribe_id == subscription_id){
    if (hex_flag == 1) {
    for (i = 0; i < message->data_length; i++) {
    printf("%02x", message->data[i]);
    }
    printf("\n");
    } else {
    printf("%s\n", message->data);
    }
    if(print_metadata_flag == 1 && choir_map_length(message->metadata)>0){
    printf("\nMetadata : \n");
    choir_map_print(message->metadata);
    printf("\n");
    }
    }else{
    printf("Receive distribute message with incorrect subscription id\n");
    }
    (void)u_data;
    fflush(stdout);
    }
    int main(int argc, char* const* argv)
    {
    choir_socket_handler_t* choir_handler;
    int subscribe_flag = 0, publish_flag = 0, binary_flag = 0, application_id_flag = 0, message_id_flag = 0,
    reliable_flag = 0;
    int64_t message_id = 0;
    int64_t application_id = 0;
    uint8_t pubsub_format = CHOIR_MESSAGE_FORMAT_GENERIC;
    const char *metadata_list[DEFAULT_ENTRY_NUMBER] = {};
    int metadata_list_len = 0;
    int n = 0; // error handler
    char connection_string[256] = { 0x0 };
    char format[] = "spa:m:d:j:hvrbRMj:S:";
    int optch;
    signal(SIGINT, intHandler);
    char* end = NULL;
    setDebug(2);
    opterr = 1;
    if(getenv("CHOIR_SERVER_URL") != NULL){
    strcpy(connection_string,getenv("CHOIR_SERVER_URL"));
    }else{
    strcpy(connection_string, "tcp://127.0.0.1:49154");
    }
    while ((optch = getopt(argc, (char* const*)argv, format)) != -1)
    switch (optch) {
    case 'a':
    application_id = strtol(optarg, &end, 10);
    if (*end){
    printf("Application id argument error: %s\n", end);
    }else if(application_id < 0 || application_id > (int64_t)UINT32_MAX){
    printf("Application id should be comprised between [%u,%"PRIu32"]\n",0,UINT32_MAX);
    }else{
    application_id_flag = 1;
    }
    break;
    case 'm':
    message_id = strtol(optarg, &end, 10);
    if (*end){
    printf("Message id argument error: %s\n", end);
    }else if(message_id < 0 || message_id > (int32_t)UINT16_MAX){
    printf("Message id should be comprised between [%u,%"PRIu16"]\n",0,UINT16_MAX);
    }else{
    message_id_flag = 1;
    }
    break;
    case 'j':
    pubsub_format = strtol(optarg, &end, 10);
    if (*end){
    printf("pubsub_format argument error: %s\n", end);
    exit(1);
    }
    break;
    case 'd':
    strcpy(connection_string, optarg);
    break;
    case 's':
    subscribe_flag = 1;
    if (publish_flag == 1)
    publish_flag = 0;
    break;
    case 'p':
    publish_flag = 1;
    if (subscribe_flag == 1)
    subscribe_flag = 0;
    break;
    case 'r':
    hex_flag = 1;
    break;
    case 'b':
    binary_flag = 1;
    break;
    case 'R':
    reliable_flag = 1;
    break;
    case 'M':
    print_metadata_flag = 1;
    break;
    case 'S':
    metadata_list[metadata_list_len++] = optarg;
    break;
    case 'h':
    print_help();
    break;
    case 'v':
    setDebug(0);
    break;
    default:
    print_help();
    }
    if (publish_flag == 0 && subscribe_flag == 0)
    print_help();
    if (message_id_flag == 1 && application_id_flag == 1) {
    choir_handler = choir_alloc_with_size(BUFFER_LENGTH);
    if (choir_handler == NULL) {
    logError(TAG, "choir_alloc failed");
    exit(1);
    }
    n = choir_set_timeout(choir_handler, 1000);
    if(n){
    logError(TAG, "choir_set_timeout failed");
    choir_free(choir_handler);
    exit(1);
    }
    // Generic connect to server (TCP/UDP)
    n = choir_connect(choir_handler, connection_string, application_id);
    if (n == -1) {
    logError(TAG, "choir_connect failed");
    choir_free(choir_handler);
    exit(1);
    }
    if (publish_flag == 1) {
    // Alloc choir_socket_handler_t structure
    char* read;
    struct choir_pubsub_message message = CHOIR_PUBSUB_MESSAGE_INIT;
    void** key_array = calloc(metadata_list_len, sizeof(void*));
    void** value_array = calloc(metadata_list_len, sizeof(void*));
    if (metadata_list_len > 0) {
    message.metadata = mapentry_cache_put_from_args(metadata_list, metadata_list_len, 1,
    key_array, value_array);
    }
    int binary_pos = 0;
    while (n != -1 && keepRunning) {
    message.data = malloc(sizeof(char) * BUFFER_LENGTH);
    if (message.data == NULL) {
    printf("malloc error");
    break;
    }
    if (binary_flag) {
    if (binary_pos) {
    read = NULL;
    } else {
    int tmpchr = 0;
    int16_t tmp_complement = 0;
    while ((tmpchr = fgetc(stdin)) != EOF) {
    if((tmp_complement = fgetc(stdin)) != EOF){
    tmpchr |= tmp_complement<<8;
    message.data[binary_pos] = (char)strtoul((const char*)&tmpchr,NULL,16);
    binary_pos++;
    }
    }
    read = (char*)message.data;
    }
    } else {
    read = fgets((char*)message.data, BUFFER_LENGTH,
    stdin); // getline((char **) &line, &len, stdin);
    }
    if (read == NULL) {
    free(message.data);
    // perform disconnect if needed and release socket
    choir_disconnect(choir_handler);
    // free allocated memory for choir_handler
    choir_free(choir_handler);
    int err = errno;
    logError(TAG, strerror(err));
    return -1;
    }
    if ((void*) read != NULL) {
    if (binary_flag) {
    message.data_length = binary_pos;
    } else {
    message.data_length = strlen((char*) message.data);
    }
    if (pubsub_format) {
    message.format = pubsub_format;
    }
    if (reliable_flag) {
    struct choir_pubsub_message *message_resp = NULL;
    uint16_t nbr_response;
    n = choir_request(choir_handler, message_id, &message, &nbr_response, &message_resp);
    if (nbr_response > 0) {
    int i = 0;
    for (i = 0; i < nbr_response; i++) {
    printf("%.*s\n", message_resp[i].data_length, message_resp[i].data);
    free(message_resp[i].data);
    }
    free(message_resp);
    }
    }else {
    n = choir_publish(choir_handler, message_id, &message);
    }
    }
    usleep(50000); // todo
    // publish string to server by corresponding message ID
    free(message.data);
    message.data = NULL;
    if (feof(stdin)) {
    break;
    }
    }
    if(message.data != NULL){
    free(message.data);
    }
    mapentry_free_array(key_array, metadata_list_len);
    mapentry_free_array(value_array, metadata_list_len);
    free(key_array);
    free(value_array);
    } else if (subscribe_flag == 1) {
    // Subscribe to corresponding MESSAGE_ID and register callback function
    // "print_distribute_message" that will be called when receiving distribute message
    // return a subscribe_id use when you want to unsubscribe from server
    subscribe_id = choir_subscribe_with_format(choir_handler, message_id, pubsub_format, NULL, print_distribute_message,NULL);
    choir_loop_set_timeout(choir_handler, 200);
    if (subscribe_id != -1) {
    while (n >=0 && keepRunning) {
    n = choir_loop_once(choir_handler);
    }
    // check distribute message that correspond to MESSAGE_ID and give them to callback
    // function "print_distribute_message"
    // Should never return
    // Unsubscribe from MESSAGE_ID, server should stop to distribute message to this
    // APPLICATION_ID
    n = choir_unsubscribe(choir_handler, subscribe_id);
    }
    }
    // perform disconnect if needed and release socket
    choir_disconnect(choir_handler);
    // free allocated memory for choir_handler
    choir_free(choir_handler);
    }
    return 0;
    }
