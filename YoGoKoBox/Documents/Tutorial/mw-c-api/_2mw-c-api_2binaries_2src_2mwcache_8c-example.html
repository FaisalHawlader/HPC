    static void print_cache_object_json(const struct choir_map* cache_object)
    {
    uint64_t i = 0, j = 0;
    printf("{");
    for (i = 0; i < choir_map_length(cache_object); i++) {
    const char* key = choir_map_key(cache_object, i);
    uint8_t type = choir_map_type(cache_object, key);
    size_t size = 0;
    if (i > 0) {
    printf(",");
    }
    const uint8_t* ptr = NULL;
    switch (type) {
    case MAP_TYPE_NULL:
    printf("\"%s\":null", key);
    break;
    case MAP_TYPE_BOOLEAN:
    printf("\"%s\":%s", key, choir_map_get_bool(cache_object, key, false) ? "true" : "false");
    break;
    case MAP_TYPE_INT:
    printf("\"%s\":%"PRId64, key, choir_map_get_int(cache_object, key, 0));
    break;
    case MAP_TYPE_DOUBLE:
    printf("\"%s\":%f", key, choir_map_get_double(cache_object, key, 0.0));
    break;
    case MAP_TYPE_STRING:
    size = 0;
    ptr = (const uint8_t*)choir_map_get_string(cache_object, key, "empty", &size);
    printf("\"%s\":\"%.*s\"", key, (int)size, ptr);
    break;
    case MAP_TYPE_BIN:
    size = 0;
    ptr = choir_map_get_bin(cache_object, key, 0, &size);
    printf("\"%s\":[", key);
    for (j = 0; j < size; j++) {
    printf("%02x%c", ptr[j], (j < size - 1) ? ',' : ']');
    }
    break;
    }
    }
    printf("}\n");
    }
    static void on_notify_function(uint8_t update_type, int64_t object_id, int nbr_key, const char** key_changed_list,
    const struct choir_map* cache_object, void* u_data)
    {
    printf("receive update type : %s for object id : %" PRId64 " and key : \n", get_cache_update_type(update_type),
    object_id);
    uint8_t print_format = *(uint8_t*)u_data;
    int i = 0;
    if (nbr_key > 0) {
    printf("[");
    for (i = 0; i < nbr_key; i++) {
    printf("\"%s\"", key_changed_list[i]);
    if (i != nbr_key - 1) {
    printf(",");
    } else {
    printf("]\n");
    }
    }
    }
    if (cache_object && choir_map_length(cache_object) > 0) {
    if (print_format == 0) {
    choir_map_print(cache_object);
    } else if (print_format == 1) {
    print_cache_object_json(cache_object);
    }
    printf("\n");
    }
    fflush(stdout);
    }
    static int set_map_from_args(choir_socket_handler_t* choir_handler, uint32_t cache_id, int64_t object_id,
    char** extra_args, uint32_t nbr_extra_args, uint8_t escape_flag, uint8_t replace)
    {
    struct choir_map* c_object = NULL;
    void** key_array = calloc(nbr_extra_args, sizeof(void*));
    void** value_array = calloc(nbr_extra_args, sizeof(void*));
    uint32_t i = 0;
    int n = 0;
    c_object = mapentry_cache_put_from_args(extra_args, nbr_extra_args, escape_flag,
    key_array, value_array);
    if (c_object == NULL ) {
    logError(TAG, "Args parsing error\n");
    goto free;
    }
    if (nbr_extra_args > 0) {
    if (replace == 0) {
    n = choir_cache_set(choir_handler, cache_id, object_id, c_object);
    } else if (replace == 1) {
    n = choir_cache_replace(choir_handler, cache_id, object_id, c_object);
    }
    printf("%s\n", get_cache_set_status(n));
    }
    free:
    mapentry_free_array(key_array, nbr_extra_args);
    mapentry_free_array(value_array, nbr_extra_args);
    free(key_array);
    free(value_array);
    choir_map_free(c_object);
    return -1;
    }
    static int set_cache_from_file(choir_socket_handler_t* choir_handler, uint32_t cache_id, int64_t object_id, FILE* file,
    uint8_t escape_flag, uint8_t replace)
    {
    char buffer[BUFFER_LENGTH];
    int buflen = 0;
    struct choir_map* c_object = choir_map_init(DEFAULT_ENTRY_NUMBER);
    void** key_array = calloc(DEFAULT_ENTRY_NUMBER, sizeof(void*));
    void** value_array = calloc(DEFAULT_ENTRY_NUMBER, sizeof(void*));
    size_t cur_size = DEFAULT_ENTRY_NUMBER;
    int n = 0;
    size_t entry_number = 0;
    uint32_t i = 0;
    while (fgets(buffer, BUFFER_LENGTH, file)) {
    buflen = strlen(buffer);
    if (buffer[buflen-1] == '\n') {
    buffer[buflen-1] = '\0';
    }
    if (entry_number == cur_size) {
    key_array = realloc(key_array, cur_size * 2);
    value_array = realloc(value_array, cur_size * 2);
    c_object = choir_map_expand(c_object, cur_size * 2);
    cur_size = cur_size * 2;
    }
    key_array[entry_number] = mapentry_extract_key(buffer);
    if (key_array[entry_number] == NULL) {
    continue;
    }
    value_array[entry_number] = mapentry_extract_value(buffer, file);
    if (value_array[entry_number] == NULL) {
    goto put_error;
    }
    if (!mapentry_cache_put(c_object, key_array[entry_number], &(value_array[entry_number]), escape_flag)) {
    goto put_error;
    }
    memset(buffer, 0x0, BUFFER_LENGTH);
    entry_number++;
    }
    if (entry_number > 0) {
    if (replace == 0) {
    i = choir_cache_set(choir_handler, cache_id, object_id, c_object);
    } else if (replace == 1) {
    i = choir_cache_replace(choir_handler, cache_id, object_id, c_object);
    }
    printf("%s\n", get_cache_set_status(i));
    }
    goto free;
    put_error:
    logError(TAG, "File parsing error\n");
    free:
    mapentry_free_array(key_array, entry_number);
    mapentry_free_array(value_array, entry_number);
    free(key_array);
    free(value_array);
    choir_map_free(c_object);
    if (ferror(file)) {
    perror("Error reading from stdin.");
    }
    return -1;
    }
    int main(int argc, char* const* argv)
    {
    choir_socket_handler_t* choir_handler;
    int application_id_flag = 0, list_flag = 0, get_flag = 0, set_flag = 0, remove_flag = 0, replace_flag = 0,
    register_flag = 0, cache_id_flag = 0, stdin_flag = 0, file_flag = 0, object_id_flag = 0;
    int64_t cache_id = 0;
    int64_t object_id = 0;
    int64_t application_id = 0;
    int n = 0;
    uint32_t i = 0, nbr_extra_args = 0, extra_args_start = 0;
    uint8_t print_format = 0;
    char key_list[512] = { 0x0 };
    char connection_string[256] = { 0x0 };
    char file_string[256] = { 0x0 };
    int64_t delay_ms = 0;
    char** extra_args = NULL;
    uint8_t escape_flag = 0;
    signal(SIGINT, intHandler);
    char* end = NULL;
    setDebug(0);
    opterr = 1;
    if (getenv("CHOIR_SERVER_URL") != NULL) {
    strcpy(connection_string, getenv("CHOIR_SERVER_URL"));
    } else {
    strcpy(connection_string, "tcp://127.0.0.1:49154");
    }
    char format[] = "vhlsgurweja:c:o:p:f:t:d:";
    int c;
    static struct option long_options[] = { /* These options set a flag. */
    { "verbose", no_argument, 0, 'v' }, { "help", no_argument, 0, 'h' }, { "list", no_argument, 0, 'l' },
    { "set", no_argument, 0, 's' }, { "get", no_argument, 0, 'g' }, { "register", no_argument, 0, 'r' },
    { "replace", no_argument, 0, 'w' }, { "remove", no_argument, 0, 'e' },
    { "application_id", required_argument, 0, 'a' }, { "cache_id", required_argument, 0, 'c' },
    { "object_id", required_argument, 0, 'o' }, { "file", required_argument, 0, 'f' },
    { "delay", required_argument, 0, 't' }, { "server", required_argument, 0, 'd' },
    { "json", no_argument, 0, 'j' }, { "unescape", no_argument, 0, 'u' }, { 0, 0, 0, 0 }
    };
    while (1) {
    /* getopt_long stores the option index here. */
    int option_index = 0;
    c = getopt_long(argc, (char* const*)argv, format, long_options, &option_index);
    /* Detect the end of the options. */
    if (c == -1) {
    break;
    }
    switch (c) {
    case 'v':
    setDebug(0);
    i++;
    break;
    case 'h':
    print_help();
    i++;
    break;
    case 'j':
    print_format = 1;
    i++;
    break;
    case 'u':
    escape_flag = 1;
    i++;
    break;
    case 'l':
    list_flag = 1;
    i++;
    break;
    case 's':
    set_flag = 1;
    i++;
    break;
    case 'g':
    get_flag = 1;
    i++;
    break;
    case 'r':
    register_flag = 1;
    i++;
    break;
    case 'w':
    replace_flag = 1;
    i++;
    break;
    case 'e':
    remove_flag = 1;
    i++;
    break;
    case 'a':
    application_id = strtol(optarg, &end, 10);
    if (*end){
    printf("Application id argument error: %s\n", end);
    }else if(application_id < 0 || application_id > (int64_t)UINT32_MAX){
    printf("Application id should be comprised between [%u,%"PRIu32"]\n",0,UINT32_MAX);
    }else{
    application_id_flag = 1;
    i += 2;
    }
    break;
    case 'c':
    cache_id = strtol(optarg, &end, 10);
    if (*end){
    printf("Cache id argument error: %s\n", end);
    }else if(cache_id < 0 || cache_id > (int64_t)UINT32_MAX){
    printf("Cache id should be comprised between [%u,%"PRIu32"]\n",0,UINT32_MAX);
    }else{
    cache_id_flag = 1;
    i += 2;
    }
    break;
    case 'o':
    object_id = strtoll(optarg, &end, 10);
    if (*end){
    printf("Cache id argument error: %s\n", end);
    }else{
    object_id_flag = 1;
    i += 2;
    }
    break;
    case 'p':
    strcpy(key_list, optarg);
    i += 2;
    break;
    case 'f':
    file_flag = 1;
    strcpy(file_string, optarg);
    i += 2;
    break;
    case 't':
    delay_ms = strtol(optarg, &end, 10);
    if (*end){
    printf("Delay ms argument error: %s\n", end);
    exit(1);
    }else if(delay_ms < 0 || delay_ms > (int64_t)UINT32_MAX){
    printf("Delay milliseconds should be comprised between [%u,%"PRIu32"]\n",0,UINT32_MAX);
    exit(1);
    }else{
    i += 2;
    }
    break;
    case 'd':
    strcpy(connection_string, optarg);
    i += 2;
    break;
    }
    }
    extra_args_start = i + 1;
    if (extra_args_start > 0) { // Get extra args
    int j = 0;
    nbr_extra_args = argc - extra_args_start;
    extra_args = (char**)malloc(sizeof(char*) * nbr_extra_args);
    if (nbr_extra_args == 1 && *argv[extra_args_start] == '-') {
    stdin_flag = 1;
    nbr_extra_args = 0;
    } else {
    for (i = extra_args_start; i < (uint32_t)argc; i++) {
    extra_args[j] = (char*)argv[i];
    j++;
    }
    }
    }
    if (application_id_flag == 1) {
    choir_handler = choir_alloc();
    if (choir_handler == NULL) {
    logError(TAG, "choir_alloc failed");
    exit(1);
    }
    n = choir_set_timeout(choir_handler, 1000);
    if(n){
    logError(TAG, "choir_set_timeout failed");
    choir_free(choir_handler);
    exit(1);
    }
    // Generic connect to server (TCP/UDP)
    n = choir_connect(choir_handler, connection_string, application_id);
    if (n == -1) {
    choir_free(choir_handler);
    exit(1);
    }
    // List request
    if (list_flag == 1 && cache_id_flag != 0) {
    uint64_t nbr_object = 0;
    int64_t* object_list = NULL;
    n = choir_cache_list(choir_handler, cache_id, &nbr_object, &object_list);
    if (nbr_object > 0) {
    printf("[");
    for (i = 0; i < nbr_object; i++) {
    printf("%" PRId64, object_list[i]);
    if (i == nbr_object - 1) {
    printf("]\n");
    } else {
    printf(",");
    }
    }
    }
    choir_cache_list_free(&object_list);
    } else if (get_flag == 1 && cache_id_flag != 0 && object_id_flag != 0) { // Get Request
    struct choir_map* cache_object =
    choir_cache_get(choir_handler, cache_id, object_id, nbr_extra_args, extra_args);
    if (print_format == 0) {
    choir_map_print(cache_object);
    } else if (print_format == 1) {
    print_cache_object_json(cache_object);
    }
    choir_map_free(cache_object);
    } else if (set_flag == 1 && cache_id_flag != 0 && object_id_flag != 0) { // Set request
    if (stdin_flag == 1) {
    set_cache_from_file(choir_handler, cache_id, object_id, stdin, escape_flag, 0);
    } else if (file_flag == 1) {
    FILE* fp;
    fp = fopen(file_string, "r");
    set_cache_from_file(choir_handler, cache_id, object_id, fp, escape_flag, 0);
    fclose(fp);
    } else if (nbr_extra_args > 0) {
    set_map_from_args(choir_handler, cache_id, object_id, extra_args, nbr_extra_args,
    escape_flag, 0);
    }
    } else if (remove_flag == 1 && cache_id_flag != 0 && object_id_flag != 0) { // Remove request
    n = choir_cache_remove(choir_handler, cache_id, object_id);
    } else if (replace_flag == 1 && cache_id_flag != 0 && object_id_flag != 0) { // Replace request
    if (stdin_flag == 1) {
    set_cache_from_file(choir_handler, cache_id, object_id, stdin, escape_flag, 1);
    } else if (file_flag == 1) {
    FILE* fp;
    fp = fopen(file_string, "r");
    set_cache_from_file(choir_handler, cache_id, object_id, fp, escape_flag, 1);
    fclose(fp);
    } else if (nbr_extra_args > 0) {
    set_map_from_args(choir_handler, cache_id, object_id, extra_args, nbr_extra_args,
    escape_flag, 1);
    }
    } else if (register_flag == 1 && cache_id_flag != 0) {
    int register_id = 0;
    choir_loop_set_timeout(choir_handler, 200);
    if (object_id_flag == 0) {
    register_id = choir_cache_register(choir_handler, cache_id, object_id, delay_ms,
    CACHE_REGISTER_ALL_OBJECT, nbr_extra_args, extra_args, &print_format,
    on_notify_function);
    } else {
    register_id = choir_cache_register(choir_handler, cache_id, object_id, delay_ms, 0,
    nbr_extra_args, extra_args, &print_format, on_notify_function);
    }
    if (register_id != -1) {
    while (n >= 0 && keepRunning) {
    n = choir_loop_once(choir_handler);
    }
    choir_cache_unregister(choir_handler, register_id);
    }
    }
    // perform disconnect if needed and release socket
    choir_disconnect(choir_handler);
    // free allocated memory for choir_handler
    choir_free(choir_handler);
    } else {
    short_help();
    }
    free(extra_args);
    return 0;
    }
